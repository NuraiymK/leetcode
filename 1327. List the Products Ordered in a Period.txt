/* https://leetcode.com/problems/list-the-products-ordered-in-a-period */

select
    p.product_name,
    unit = sum ( o.unit ) 
from
    Products p
        inner join
    Orders o on p.product_id = o.product_id
        and year ( o.order_date ) = 2020
        and month (o.order_date ) = 2
group by p.product_name
having sum ( o.unit ) >= 100
