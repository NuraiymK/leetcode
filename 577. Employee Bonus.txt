with bonus_CTE (name, bonus)
as
(
    select emp.name, bn.bonus
    from
        Employee emp
            left outer join
        Bonus bn on emp.empId = bn.empId
)
select *
from bonus_CTE
where bonus < 1000 or bonus is null
